161  def execute(self, fp):
162  # Check that joint is not controlled by a RobotPanel
163  if not fp.RobotPanelActive:
164  # Check that current translation has valid format
165  if not fp.ValidTranslation:
166  FreeCAD.Console.PrintWarning(fp.Name +
167  ".execute(): Translation " +
168  "is not in a valid format.\n")
169  return
170 
171  # Update d according to current time and translation
172  indices, weights = self.find_timestamp_indices_and_weights(fp)
173 
174  fp.d = fp.dOffset \
175  + (weights[0]*fp.dSequence[indices[0]]
176  + weights[1]*fp.dSequence[indices[1]])
177 
178  # DH transformation
179  T_theta = FreeCAD.Placement(FreeCAD.Vector(0, 0, 0),
180  FreeCAD.Rotation(FreeCAD.Vector(0, 0, 1),
181  fp.theta))
182  T_d = FreeCAD.Placement(FreeCAD.Vector(0, 0, fp.d),
183  FreeCAD.Rotation(FreeCAD.Vector(0, 0, 1), 0))
184  T_a = FreeCAD.Placement(FreeCAD.Vector(fp.a, 0, 0),
185  FreeCAD.Rotation(FreeCAD.Vector(1, 0, 0), 0))
186  T_alpha = FreeCAD.Placement(FreeCAD.Vector(0, 0, 0),
187  FreeCAD.Rotation(FreeCAD.Vector(1, 0, 0),
188  fp.alpha))
189 
190  fp.ObjectPlacement = T_theta.multiply(T_d.multiply(
191  T_a.multiply(T_alpha)))
192 
193  # Placement update
194  fp.Placement = fp.ParentFramePlacement.multiply(fp.ObjectPlacement)

452  def is_ValidTranslation(self, timestamps=[], ds=[], translation=None):
453  # Check all keys are included and record lengths of their lists
454  if translation is not None and isinstance(translation, dict):
455  for key in ["Timestamps", "dSequence"]:
456  if key not in translation.keys():
457  FreeCAD.Console.PrintWarning("Translation misses key " +
458  key + ".\n")
459  return False
460  timestamps = translation["Timestamps"]
461  ds = translation["thetaSequence"]
462 
463  # Check that all lists have the same length
464  if len(timestamps) == 0 or \
465  (len(timestamps) != 0 and len(timestamps) != len(ds)):
466  FreeCAD.Console.PrintWarning("Translation has lists with "
467  + "inconsistent or zero "
468  + "lengths.\n")
469  return False
470 
471  # Check timestamps correspond to list of increasing values
472  if any([timestamps[i] >= timestamps[i+1]
473  for i in range(len(timestamps)-1)]):
474  FreeCAD.Console.PrintWarning("Translation 'Timestamps' is not "
475  + "list of increasing values.\n")
476  return False
477 
478  if not all([self.fp.dMinimum <= d <= self.fp.dMaximum for d in ds]):
479  FreeCAD.Console.PrintWarning("Translation 'dSequence' elements"
480  + " are out of d range.\n")
481  return False
482 
483  return True

505  def find_timestamp_indices_and_weights(self, fp):
506  # Retrieve indices corresponding to current time
507  # If the time is before the first Timestamp use the first Timestamp
508  if fp.Time <= fp.Timestamps[0]:
509  indices = [0, 0]
510  weights = [1, 0]
511 
512  # If the time is after the last Timpestamp use the last Timestamp
513  elif fp.Time >= fp.Timestamps[-1]:
514  indices = [-1, -1]
515  weights = [1, 0]
516 
517  # If time is in the range of Timesteps
518  else:
519  # Find the index of the closest higher value
520  indices = [bisect(fp.Timestamps, fp.Time)]
521  # Add the previous index
522  indices.insert(0, indices[0]-1)
523  weights = [fp.Timestamps[indices[1]] - fp.Time,
524  fp.Time - fp.Timestamps[indices[0]]]
525  if not fp.Interpolate:
526  if weights[0] > weights[1]:
527  weights = [1, 0]
528  else:
529  weights = [0, 1]
530  else:
531  weights = [weights[0]/sum(weights), weights[1]/sum(weights)]
532 
533  return indices, weights
