688  def makeFrame(self, frame_labels):
689  # make a generic shaft from 0 in Y direction
690  shaft_vertices = coin.SoVertexProperty()
691  shaft_vertices.vertex.setNum(2)
692  shaft_vertices.vertex.set1Value(0, 0, 0, 0)
693  self.frame_shaft = coin.SoLineSet()
694  self.frame_shaft.vertexProperty.setValue(shaft_vertices)
695  self.frame_shaft.numVertices.setNum(1)
696  self.frame_shaft.numVertices.setValue(2)
697 
698  # make a generic conic arrowhead oriented in Y axis direction and
699  # move it at the end of the shaft
700  self.frame_arrowhead_translation = coin.SoTranslation()
701  self.frame_arrowhead_cone = coin.SoCone()
702  self.frame_arrowhead = coin.SoSwitch()
703  self.frame_arrowhead.addChild(self.frame_arrowhead_translation)
704  self.frame_arrowhead.addChild(self.frame_arrowhead_cone)
705 
706  # make rotations to rotate prepared shaft and arrowhead for Y axis
707  # direction also to X and Z
708  rot_y2x = coin.SoRotation()
709  rot_y2x.rotation.setValue(coin.SbRotation(coin.SbVec3f(0, 1, 0),
710  coin.SbVec3f(1, 0, 0)))
711  rot_y2z = coin.SoRotation()
712  rot_y2z.rotation.setValue(coin.SbRotation(coin.SbVec3f(0, 1, 0),
713  coin.SbVec3f(0, 0, 1)))
714 
715  # prepare colors for X,Y,Z which will correspond to R,G,B as customary
716  self.frame_color_x = coin.SoPackedColor()
717  self.frame_color_y = coin.SoPackedColor()
718  self.frame_color_z = coin.SoPackedColor()
719 
720  # make complete colored and rotated arrows
721  x_arrow = coin.SoSeparator()
722  x_arrow.addChild(rot_y2x)
723  x_arrow.addChild(self.frame_color_x)
724  x_arrow.addChild(self.frame_shaft)
725  x_arrow.addChild(self.frame_arrowhead)
726  x_arrow.addChild(frame_labels[0])
727  y_arrow = coin.SoSeparator()
728  y_arrow.addChild(self.frame_color_y)
729  y_arrow.addChild(self.frame_shaft)
730  y_arrow.addChild(self.frame_arrowhead)
731  y_arrow.addChild(frame_labels[1])
732  z_arrow = coin.SoSeparator()
733  z_arrow.addChild(rot_y2z)
734  z_arrow.addChild(self.frame_color_z)
735  z_arrow.addChild(self.frame_shaft)
736  z_arrow.addChild(self.frame_arrowhead)
737  z_arrow.addChild(frame_labels[2])
738 
739  # prepare draw style to control shaft width
740  self.frame_drawstyle = coin.SoDrawStyle()
741 
742  # make complete frame and it to shaded display mode
743  separated_frame = coin.SoSeparator()
744  separated_frame.addChild(self.frame_drawstyle)
745  separated_frame.addChild(x_arrow)
746  separated_frame.addChild(y_arrow)
747  separated_frame.addChild(z_arrow)
748 
749  return separated_frame
