399  @Slot(float, float)
400  def play(self, t):
401  # Load current time
402  time_ = time.clock()
403 
404  # Check pause button was not pressed
405  if self.last_clicked == "pause":
406  return
407 
408  # Disribute the animation time to trajectories so that they change
409  # positions of all animated objects
410  self.distributeTime(t)
411  self.updateCollisions()
412  self.showChanges()
413 
414  # Display current progres on the seek slider
415  self.form.sld_seek.setValue(
416  numpy.round(100*(t - self.control_proxy.StartTime)
417  / (self.control_proxy.StopTime
418  - self.control_proxy.StartTime)))
419 
420  # Stop the animation if the animation time reached a range boundary
421  if t >= self.control_proxy.StopTime:
422  self.last_clicked = "pause"
423  self.setInvalidButtons()
424  return
425 
426  # Compute an animation time for the next frame
427  next_t = min(t + self.control_proxy.StepTime,
428  self.control_proxy.StopTime)
429 
430  # Compute pause period so that animaiton time roughly corresponds to
431  # the real time
432  pause = round(1000*(self.control_proxy.StepTime + time_
433  - time.clock()))
434  pause = pause*(pause > 0)
435 
436  # Setup a timer to show next frame if animaiton wasn't paused
437  if self.last_clicked != "pause":
438  self.timer.singleShot(pause, lambda: self.play(next_t))

550  def distributeTime(self, t):
551  # Load list of objects inside Control group
552  objects = self.control_proxy.Group
553 
554  # Go through them, their children and update time,
555  # if they are Trajectories
556  while len(objects) > 0:
557  obj = objects.pop(0)
558  if obj.Proxy.__class__.__name__ == "TrajectoryProxy":
559  obj.Time = t
560  objects += obj.Group

636  def findSequences(self, files):
637  # Check there are any files
638  if len(files) == 0:
639  return {}
640 
641  # Go through the files
642  sequences = {}
643  for f in files:
644 
645  # Check they fit the name pattern
646  img_name = re.search(r"(seq\d+)-(\d+)(?=\.png)", f)
647  if img_name is not None:
648 
649  # Add new sequences
650  if img_name.group(1) not in list(sequences.keys()):
651 
652  # Add sequence if it's starting with 0
653  if int(img_name.group(2)) == 0:
654  sequences[img_name.group(1)] = 1
655  last_frame = int(img_name.group(2))
656 
657  # Compute number of successive frames
658  elif int(img_name.group(2)) == (last_frame + 1):
659  sequences[img_name.group(1)] += 1
660  last_frame += 1
661 
662  # Remove sequence if a frame is missing
663  else:
664  sequences.pop(img_name.group(1))
665 
666  # Leave sequences longer than 1 frame
667  sequences = {key: val for key, val in sequences.items() if val > 1}
668  return sequences
