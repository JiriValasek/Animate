119  def slotChangedObject(self, obj, prop):
120  # If objects are added to a group
121  if prop == "Group" and len(obj.Group) > len(self.group_before):
122  # If a new object is added to a group object from Animate workbench
123  new_obj = obj.Group[-1]
124  if self.isAnimateGroup(obj):
125  # An object not from Animate workbench was added to it
126  if self.foreignObjectInAnimateGroup(new_obj, obj):
127  QMessageBox.warning(
128  None, 'Forbidden action detected',
129  "Group objects from Animate workbench can group\n"
130  + "only selected objects from Animate workbench.\n"
131  + "Check the user guide for more info.")
132  FreeCAD.ActiveDocument.undo()
133  else:
134  FreeCAD.ActiveDocument.commitTransaction()
135 
136  # An object is added to a group not from Animate workbench
137  else:
138  # The added object was from Animate workbench
139  if self.animateObjectInForeignGroup(new_obj, obj):
140  QMessageBox.warning(

150  # If objects are removed from a group

215  def foreignObjectInAnimateGroup(self, obj, group):
216  # Only a Trajectory can be in a Trajectory group
217  if group.Proxy.__class__.__name__ == "TrajectoryProxy" and \
218  hasattr(obj, "Proxy") and \
219  obj.Proxy.__class__.__name__ == "TrajectoryProxy":
220  return False
221 
222  # Only some objects can be in a Control group
223  elif group.Proxy.__class__.__name__ == "ControlProxy" and \
224  hasattr(obj, "Proxy") and \
225  obj.Proxy.__class__.__name__ in ALLOWED_IN_CONTROL:
226  return False
227 
228  elif group.Proxy.__class__.__name__ == "CollisionDetectorProxy" and \
229  obj.Name.find("Collision") != -1:
230  return False
231  return True
