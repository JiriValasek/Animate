132  def run(self):
133  # Try to connect to an incomming tcp socket using its socket descriptor
134  tcpSocket = QTcpSocket()
135  if not tcpSocket.setSocketDescriptor(self.socketDescriptor):
136  FreeCAD.Console.PrintError("Socket not accepted.\n")
137  return
138 
139  # Wait for an incomming message
140  if not tcpSocket.waitForReadyRead(msecs=WAIT_TIME_MS):
141  FreeCAD.Console.PrintError("No request send.\n")
142  return
143 
144  # Make an input data stream
145  instr = QDataStream(tcpSocket)
146  instr.setVersion(QDataStream.Qt_4_0)
147 
148  # Try to read the message size
149  if self.blockSize == 0:
150  if tcpSocket.bytesAvailable() < 2:
151  return
152  self.blockSize = instr.readUInt16()
153 
154  # Check message is sent complete
155  if tcpSocket.bytesAvailable() < self.blockSize:
156  return
157 
158  # Read message and inform about it
159  instr = instr.readString()
160  FreeCAD.Console.PrintLog("CommandServer received> "
161  + str(instr) + "\n")
162 
163  # Try to execute the message string and prepare a response
164  try:
165  exec(str(instr))
166  except Exception as e:
167  FreeCAD.Console.PrintError("Executing external command failed:"
168  + str(e) + "\n")
169  message = "Command failed - " + str(e)
170  else:
171  FreeCAD.Console.PrintLog("Executing external command succeded!\n")
172  message = COMMAND_EXECUTED_CONFIRMATION_MESSAGE
173 
174  # Prepare the data block to send back and inform about it
175  FreeCAD.Console.PrintLog("CommandServer sending> " + message + " \n")
176  block = QByteArray()
177  outstr = QDataStream(block, QIODevice.WriteOnly)
178  outstr.setVersion(QDataStream.Qt_4_0)
179  outstr.writeUInt16(0)
180  outstr.writeQString(message)
181  outstr.device().seek(0)
182  outstr.writeUInt16(block.size() - 2)
183 
184  # Send the block, disconnect from the socket and terminate the QThread
185  tcpSocket.write(block)
186  tcpSocket.disconnectFromHost()
187  tcpSocket.waitForDisconnected()

484  return CLIENT_ERROR_NO_CONNECTION

487  block = QByteArray()

502  return CLIENT_ERROR_NO_RESPONSE
