683  def is_ValidTrajectory(self, timestamps=[], translation_x=[],
684  translation_y=[], translation_z=[],
685  rotation_point_x=[], rotation_point_y=[],
686  rotation_point_z=[], rotation_axis_x=[],
687  rotation_axis_y=[], rotation_axis_z=[],
688  rotation_angle=[], trajectory=None):
689  # Check all keys are included and record lengths of their lists
690  if trajectory is not None and isinstance(trajectory, dict):
691  for key in ["Timestamps", "TranslationX", "TranslationY",
692  "TranslationZ", "RotationPointX", "RotationPointY",
693  "RotationPointZ", "RotationAxisX", "RotationAxisY",
694  "RotationAxisZ", "RotationAngle"]:
695  if key not in trajectory.keys():
696  FreeCAD.Console.PrintWarning("Trajectory misses key " +
697  key + ".\n")
698  return False
699  timestamps = trajectory["Timestamps"]
700  translation_x = trajectory["TranslationX"]
701  translation_y = trajectory["TranslationY"]
702  translation_z = trajectory["TranslationZ"]
703  rotation_point_x = trajectory["RotationPointX"]
704  rotation_point_y = trajectory["RotationPointY"]
705  rotation_point_z = trajectory["RotationPointZ"]
706  rotation_axis_x = trajectory["RotationAxisX"]
707  rotation_axis_y = trajectory["RotationAxisY"]
708  rotation_axis_z = trajectory["RotationAxisZ"]
709  rotation_angle = trajectory["RotationAngle"]
710 
711  # Check that all lists have the same length
712  if len(timestamps) == 0 or \
713  (len(timestamps) != 0 and
714  (len(timestamps) != len(timestamps) or
715  len(timestamps) != len(translation_x) or
716  len(timestamps) != len(translation_y) or
717  len(timestamps) != len(translation_z) or
718  len(timestamps) != len(rotation_point_x) or
719  len(timestamps) != len(rotation_point_y) or
720  len(timestamps) != len(rotation_point_z) or
721  len(timestamps) != len(rotation_axis_x) or
722  len(timestamps) != len(rotation_axis_y) or
723  len(timestamps) != len(rotation_axis_z) or
724  len(timestamps) != len(rotation_angle))):
725  FreeCAD.Console.PrintWarning("Trajectory has lists with "
726  + "inconsistent or zero "
727  + "lengths.\n")
728  return False
729 
730  # Check timestamps correspond to list of increasing values
731  if any([timestamps[i] >= timestamps[i+1]
732  for i in range(len(timestamps)-1)]):
733  FreeCAD.Console.PrintWarning("Trajectory 'Timestamps' is not "
734  + "list of increasing values.\n")
735  return False
736 
737  if any([sum([rotation_axis_x[i]**2,
738  rotation_axis_y[i]**2,
739  rotation_axis_z[i]**2]) != 1
740  for i in range(len(rotation_axis_x))]):
741  FreeCAD.Console.PrintWarning("Trajectory 'Rotation Axis' "
742  + "elements don't have norm 1.\n")
743  return False
744 
745  return True

767  def find_timestamp_indices_and_weights(self, fp):
768  # Retrieve indices corresponding to current time
769  # If the time is before the first Timestamp use the first Timestamp
770  if fp.Time <= fp.Timestamps[0]:
771  indices = [0, 0]
772  weights = [1, 0]
773 
774  # If the time is after the last Timpestamp use the last Timestamp
775  elif fp.Time >= fp.Timestamps[-1]:
776  indices = [-1, -1]
777  weights = [1, 0]
778 
779  # If time is in the range of Timesteps
780  else:
781  # Find the index of the closest higher value
782  indices = [bisect(fp.Timestamps, fp.Time)]
783  # Add the previous index
784  indices.insert(0, indices[0]-1)
785  weights = [fp.Timestamps[indices[1]] - fp.Time,
786  fp.Time - fp.Timestamps[indices[0]]]
787  if not fp.Interpolate:
788  if weights[0] > weights[1]:
789  weights = [1, 0]
790  else:
791  weights = [0, 1]
792  else:
793  weights = [weights[0]/sum(weights), weights[1]/sum(weights)]
794 
795  return indices, weights
